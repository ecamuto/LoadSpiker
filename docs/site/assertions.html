Assertions

Validate individual responses and aggregate performance metrics with LoadSpiker's assertion system.

Overview

LoadSpiker provides two categories of assertions:

  • Response Assertions (loadspiker.assertions) — Validate individual HTTP responses: status codes, body content, headers, JSON values.
  • Performance Assertions (loadspiker.performance_assertions) — Validate aggregate test results: throughput, error rate, response time percentiles.
from loadspiker import Engine, Scenario
from loadspiker.assertions import status_is, body_contains, json_path, run_assertions
from loadspiker.performance_assertions import (
    throughput_at_least, error_rate_below, run_performance_assertions
)

engine = Engine()
scenario = Scenario("Assertion Demo")
scenario.get("https://httpbin.org/get")

# Run the load test
results = engine.run_scenario(scenario, users=10, duration=30)

# Check response (using a single recorded response)
response = engine.execute_request("https://httpbin.org/get")
passed, failures = run_assertions(response, [
    status_is(200),
    body_contains('"url"'),
    json_path("url", expected_value="https://httpbin.org/get"),
])

# Check aggregate performance
perf_passed, perf_failures = run_performance_assertions(results, [
    throughput_at_least(5.0),
    error_rate_below(0.01),
])

Response Assertions

Import all response assertions from loadspiker.assertions.

status_is

status_is(expected_status)

Assert that the response HTTP status code equals the expected value.

ParameterTypeDescription
expected_statusintExpected HTTP status code (e.g. 200, 201, 404)
 
from loadspiker.assertions import status_is

assertion = status_is(200)
passed, message = assertion(response)
# passed = True if response['status_code'] == 200

response_time_under

response_time_under(max_ms)

Assert that the response time is below the given threshold.

ParameterTypeDescription
max_msfloatMaximum acceptable response time in milliseconds
 
from loadspiker.assertions import response_time_under

assertion = response_time_under(500)  # must respond in under 500ms
passed, message = assertion(response)

body_contains

body_contains(substring, case_sensitive=True)

Assert that the response body contains the given substring.

from loadspiker.assertions import body_contains

assertion = body_contains('"status": "ok"')
passed, message = assertion(response)

body_matches

body_matches(pattern, flags=0)

Assert that the response body matches a regular expression pattern.

from loadspiker.assertions import body_matches
import re

assertion = body_matches(r'"id":\s*\d+')
passed, message = assertion(response)

# Case-insensitive
assertion2 = body_matches(r'success', flags=re.IGNORECASE)
passed2, _ = assertion2(response)

json_path

json_path(path, expected_value=None, exists=True)

Assert a value at a dot-notation JSON path in the response body.

ParameterTypeDescription
pathstrDot-notation path (e.g. "data.user.id")
expected_valueanyExpected value at the path. If None, only existence is checked.
existsboolSet to False to assert the path does NOT exist.
Path Syntax: Use dot notation to navigate nested objects ("user.address.city") and bracket notation for arrays ("items[0].name"). The root of the parsed JSON is the starting point. 
from loadspiker.assertions import json_path

# Check specific value
assertion = json_path("data.status", expected_value="active")
passed, message = assertion(response)

# Check existence only (don't care about the value)
assertion2 = json_path("data.user.id", exists=True)

# Check that a key does NOT exist
assertion3 = json_path("data.error", exists=False)

# Array indexing
assertion4 = json_path("results[0].name", expected_value="Alice")

header_exists

header_exists(header_name, expected_value=None)

Assert that the response contains a header, optionally with a specific value.

from loadspiker.assertions import header_exists

# Check existence
assertion = header_exists("Content-Type")
passed, _ = assertion(response)

# Check value (case-insensitive value match)
assertion2 = header_exists("Content-Type", "application/json")
passed2, _ = assertion2(response)

custom_assertion

custom_assertion(fn, description=None)

Create an assertion from any callable that receives the response dict and returns a boolean.

from loadspiker.assertions import custom_assertion
import json

# Lambda assertion
assert_large_body = custom_assertion(
    lambda r: len(r['body']) > 100,
    description="Body must be over 100 bytes"
)

# Full function for complex logic
def has_valid_user(response):
    try:
        data = json.loads(response['body'])
        return (
            isinstance(data.get('id'), int) and
            isinstance(data.get('email'), str) and
            '@' in data['email']
        )
    except Exception:
        return False

assert_valid_user = custom_assertion(has_valid_user, "Response must contain a valid user object")
passed, message = assert_valid_user(response)

AssertionGroup

Combine multiple assertions with explicit AND or OR logic.

from loadspiker.assertions import AssertionGroup, status_is, body_contains, response_time_under

# AND logic — all must pass
all_pass = AssertionGroup([
    status_is(200),
    body_contains('"success"'),
    response_time_under(1000),
], logic="AND")

# OR logic — at least one must pass
any_pass = AssertionGroup([
    status_is(200),
    status_is(201),  # Accept both 200 and 201
], logic="OR")

passed, failures = all_pass.evaluate(response)

run_assertions

run_assertions(response, assertions, fail_fast=False)

Evaluate a list of assertions against a response and return a summary.

from loadspiker import Engine
from loadspiker.assertions import (
    status_is, body_contains, json_path,
    header_exists, response_time_under, run_assertions
)

engine = Engine()
response = engine.execute_request("https://api.example.com/users/1")

passed, failures = run_assertions(response, [
    status_is(200),
    response_time_under(500),
    header_exists("Content-Type", "application/json"),
    json_path("id", exists=True),
    json_path("name", expected_value="Alice"),
    body_contains('"active": true'),
])

if passed:
    print("All assertions passed!")
else:
    for failure in failures:
        print(f"FAIL: {failure}")

Performance Assertions

Import performance assertions from loadspiker.performance_assertions. These operate on the dict returned by engine.run_scenario() or engine.get_metrics().

throughput_at_least

throughput_at_least(min_rps)

Assert that the test achieved at least min_rps requests per second.

from loadspiker.performance_assertions import throughput_at_least

assertion = throughput_at_least(100.0)  # require 100+ RPS
passed, message = assertion(results)

avg_response_time_under

avg_response_time_under(max_ms)

Assert that the average response time is below the threshold.

from loadspiker.performance_assertions import avg_response_time_under

assertion = avg_response_time_under(200)  # average must be under 200ms
passed, message = assertion(results)

error_rate_below

error_rate_below(max_rate)

Assert that the error rate (fraction of failed requests) is below the threshold. Value is a fraction between 0.0 and 1.0.

from loadspiker.performance_assertions import error_rate_below

assertion = error_rate_below(0.01)  # less than 1% errors
passed, message = assertion(results)

success_rate_at_least

success_rate_at_least(min_rate)

Assert that the success rate (fraction of successful requests) meets the minimum.

from loadspiker.performance_assertions import success_rate_at_least

assertion = success_rate_at_least(0.99)  # at least 99% success
passed, message = assertion(results)

max_response_time_under

max_response_time_under(max_ms)

Assert that no single request exceeded the given response time. Useful for catching outliers.

from loadspiker.performance_assertions import max_response_time_under

assertion = max_response_time_under(5000)  # no request took over 5 seconds
passed, message = assertion(results)

run_performance_assertions

run_performance_assertions(metrics, assertions)

Evaluate a list of performance assertions against a metrics dict.

from loadspiker import Engine, Scenario
from loadspiker.performance_assertions import (
    throughput_at_least,
    avg_response_time_under,
    max_response_time_under,
    error_rate_below,
    success_rate_at_least,
    run_performance_assertions
)

engine = Engine(max_connections=200, worker_threads=8)
scenario = Scenario("SLA Validation Test")
scenario.get("https://api.example.com/products")
scenario.get("https://api.example.com/users")

results = engine.run_scenario(scenario, users=100, duration=120)

passed, failures = run_performance_assertions(results, [
    throughput_at_least(50.0),         # at least 50 RPS
    avg_response_time_under(300),       # avg < 300ms
    max_response_time_under(10000),     # no request > 10s
    error_rate_below(0.02),            # less than 2% errors
    success_rate_at_least(0.98),        # at least 98% success
])

if passed:
    print("SLA PASSED: All performance assertions satisfied.")
else:
    print("SLA FAILED:")
    for failure in failures:
        print(f"  - {failure}")
    raise SystemExit(1)  # Non-zero exit for CI pipelines
