HTTP / HTTPS

HTTP and HTTPS are fully implemented via libcurl. All HTTP methods are supported with full header and body control.

from loadspiker import Engine, Scenario

engine = Engine(max_connections=200, worker_threads=8)

# GET request
response = engine.execute_request("https://api.example.com/users")
print(response['status_code'])      # 200
print(response['body'])             # JSON string
print(response['response_time_ms']) # 42.1

# POST with JSON body
response = engine.execute_request(
    "https://api.example.com/users",
    method="POST",
    headers={
        "Content-Type": "application/json",
        "Authorization": "Bearer my-token"
    },
    body='{"name": "Alice", "email": "alice@example.com"}'
)

# PUT, DELETE
response = engine.execute_request("https://api.example.com/users/1",
                                   method="PUT",
                                   body='{"name": "Alice Updated"}')
response = engine.execute_request("https://api.example.com/users/1",
                                   method="DELETE")

Response Dictionary Keys

KeyTypeDescription
status_codeintHTTP status code (200, 404, 500, etc.)
bodystrResponse body as a string
headersdictResponse headers as key-value pairs
response_time_msfloatTotal response time in milliseconds
errorstr or NoneError message if the request failed, else None

Scenario-Based HTTP Testing

scenario = Scenario("HTTP API Test", base_url="https://api.example.com")
scenario.get("/users", headers={"Accept": "application/json"})
scenario.post("/users",
              body='{"name": "Bob"}',
              headers={"Content-Type": "application/json"})
scenario.get("/users/1")
scenario.delete("/users/1")

results = engine.run_scenario(scenario, users=50, duration=60)

WebSocket

Phase 1 Simulation: WebSocket in the current release measures connection overhead and timing without requiring libwebsockets. Real message exchange over a live WS server will be available in Phase 2. The API surface is identical to the planned final API, so your test code will not need to change. 
from loadspiker import Engine

engine = Engine(max_connections=50, worker_threads=4)

# Connect
ws_id = engine.websocket_connect(
    "wss://echo.example.com/ws",
    headers={"Authorization": "Bearer token123"}
)

# Send a message
engine.websocket_send(ws_id, '{"type": "subscribe", "channel": "prices"}')

# Send more messages
engine.websocket_send(ws_id, '{"type": "ping"}')

# Close
engine.websocket_close(ws_id)

WebSocket in a Scenario

from loadspiker.scenarios import WebSocketScenario

ws_scenario = WebSocketScenario("Chat Load Test",
                                 broker_url="wss://chat.example.com/ws")
ws_scenario.connect()
ws_scenario.send('{"type": "join", "room": "general"}')
ws_scenario.send('{"type": "message", "text": "Hello!"}')
ws_scenario.think_time(1.0)
ws_scenario.disconnect()

results = engine.run_scenario(ws_scenario, users=100, duration=60)

TCP

C Extension Required: TCP uses socket-based bindings in tcp.c. If these bindings are not yet exposed in the Python extension, calls will raise AttributeError. Check availability with the helper below before using in tests. 
from loadspiker import Engine

engine = Engine()

# Check availability
if hasattr(engine, 'tcp_connect'):
    # Connect to a raw TCP server
    tcp_id = engine.tcp_connect("tcp.example.com", 9000, timeout_ms=5000)

    # Send data (string or bytes)
    engine.tcp_send(tcp_id, "HELLO\r\n")

    # Receive response
    data = engine.tcp_receive(tcp_id, buffer_size=1024)
    print(data)

    # Disconnect
    engine.tcp_disconnect(tcp_id)
else:
    print("TCP bindings not available in this build")

TCP Scenario

from loadspiker.scenarios import TCPScenario

tcp_scenario = TCPScenario("TCP Echo Test",
                            host="echo.example.com",
                            port=9000)
tcp_scenario.connect()
tcp_scenario.send("PING\r\n")
tcp_scenario.receive(buffer_size=64)
tcp_scenario.disconnect()

results = engine.run_scenario(tcp_scenario, users=20, duration=30)

UDP

from loadspiker import Engine

engine = Engine()

# Create a UDP endpoint (port 0 = OS-assigned)
ep_id = engine.udp_create_endpoint(local_port=0)

# Send a datagram
engine.udp_send(ep_id, "metrics.example.com", 8125, "page.load:1|c")

# Receive a response (with 500ms timeout)
response = engine.udp_receive(ep_id, buffer_size=512, timeout_ms=500)
if response:
    print(f"From {response['source_host']}:{response['source_port']}")
    print(f"Data: {response['data']}")

# Close the endpoint
engine.udp_close_endpoint(ep_id)

StatsD-Style Load Test

from loadspiker import Engine, Scenario

engine = Engine()
ep_id = engine.udp_create_endpoint()

scenario = Scenario("StatsD Metrics Flood")
# Custom request logic using UDP
for i in range(10):
    engine.udp_send(ep_id, "statsd.example.com", 8125,
                    f"request.count:1|c|#env:load-test")

results = engine.run_scenario(scenario, users=50, duration=30)
engine.udp_close_endpoint(ep_id)

MQTT

MQTT is implemented via a real CONNECT/PUBLISH/SUBSCRIBE/DISCONNECT packet flow in mqtt.c. Use it to load test IoT brokers such as Mosquitto, EMQX, and HiveMQ.

MQTTScenario (Recommended)

from loadspiker.scenarios import MQTTScenario

# Basic publish scenario
mqtt_scenario = MQTTScenario(
    "IoT Temperature Publish",
    broker_host="mqtt.example.com",
    broker_port=1883
)
mqtt_scenario.connect(
    client_id="sensor-{user_id}",
    username="iot_user",
    password="secret"
)
mqtt_scenario.publish("sensors/{user_id}/temperature", payload="22.5", qos=0)
mqtt_scenario.publish("sensors/{user_id}/humidity",    payload="65",   qos=1)
mqtt_scenario.disconnect()

results = engine.run_scenario(mqtt_scenario, users=100, duration=60)
# Burst publish scenario
burst_scenario = MQTTScenario("MQTT Burst", broker_host="mqtt.example.com")
burst_scenario.connect(client_id="burst-{user_id}")
for i in range(20):
    burst_scenario.publish(f"test/topic/{i}", payload=f"message-{i}", qos=0)
burst_scenario.disconnect()
# Topic pattern scenario
pattern_scenario = MQTTScenario("MQTT Topic Pattern", broker_host="mqtt.example.com")
pattern_scenario.connect(client_id="pattern-{user_id}")
pattern_scenario.subscribe("alerts/#", qos=1)
pattern_scenario.publish("sensors/data", payload='{"temp":25}', qos=0)
pattern_scenario.disconnect()

Direct MQTT API

from loadspiker import Engine

engine = Engine()

mqtt_id = engine.mqtt_connect(
    broker_host="mqtt.example.com",
    broker_port=1883,
    client_id="loadspiker-test",
    username="user",
    password="pass",
    keepalive=60
)

engine.mqtt_publish(mqtt_id, "test/load", '{"value": 42}', qos=0)
engine.mqtt_publish(mqtt_id, "test/load", '{"value": 43}', qos=1, retain=True)
engine.mqtt_subscribe(mqtt_id, "test/responses/#", qos=0)

engine.mqtt_disconnect(mqtt_id)

QoS Levels

QoSNameDelivery GuaranteeUse Case
0 At most once Fire and forget — no acknowledgement High-frequency sensor data where some loss is acceptable
1 At least once Acknowledged delivery — may duplicate Most IoT use cases, general telemetry
2 Exactly once Four-way handshake — no duplicates, no loss Critical commands, financial events

Database

Simulation Notice: Database support in Phase 1 simulates connection pool overhead and query timing without real database drivers. This measures your application's connection handling and pool exhaustion behavior under load. Real driver integration (MySQL, PostgreSQL, MongoDB) is planned for Phase 2.

Connection Strings

DatabaseFormat
MySQLmysql://user:pass@host:3306/dbname
PostgreSQLpostgresql://user:pass@host:5432/dbname
MongoDBmongodb://user:pass@host:27017/dbname
 
from loadspiker import Engine

engine = Engine()

# MySQL connection pool
mysql_id = engine.database_connect(
    "mysql://app_user:password@db.example.com:3306/production",
    pool_size=20
)

# Execute queries
result = engine.database_query(mysql_id, "SELECT COUNT(*) FROM users")
print(result['status_code'])     # 200 = success
print(result['response_time_ms']) # simulated query latency

result = engine.database_query(
    mysql_id,
    "SELECT id, name FROM products WHERE category = ?",
    params=["electronics"]
)

engine.database_disconnect(mysql_id)

# PostgreSQL
pg_id = engine.database_connect(
    "postgresql://user:pass@pg.example.com:5432/analytics",
    pool_size=10
)
engine.database_query(pg_id, "SELECT * FROM events LIMIT 100")
engine.database_disconnect(pg_id)

# MongoDB
mongo_id = engine.database_connect(
    "mongodb://user:pass@mongo.example.com:27017/logs",
    pool_size=5
)
engine.database_query(mongo_id, '{"collection": "events", "find": {}}')
engine.database_disconnect(mongo_id)
